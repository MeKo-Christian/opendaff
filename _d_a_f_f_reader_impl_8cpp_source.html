341int DAFFReaderImpl::loadFileHeader()
342{
343 /*
344 * 1st step: Load the file header and validate it for correctness
345 */
346
347 m_fileHeader.fixEndianness();
348
349 // Check signature
350 if (!((m_fileHeader.pcSignature[0] == 'F') && (m_fileHeader.pcSignature[1] == 'W'))) {
351 // File is not an OpenDAFF database.
352 tidyup();
353 return DAFF_FILE_INVALID;
354 }
355
356 // Check version
357 // [This implementation supports version number 1.7]
358 if (m_fileHeader.iFileFormatVersion != 170) {
359 // Database version not supported
360 tidyup();
361 return DAFF_FILE_FORMAT_VERSION_UNSUPPORTED;
362 }
363
364 // Check file block entries
365 if (m_fileHeader.iNumFileBlocks <= 0) {
366 // There must be at least one block
367 tidyup();
368 return DAFF_FILE_INVALID;
369 }
370
371 return DAFF_NO_ERROR;
372}

400int DAFFReaderImpl::loadMainHeader()
401{
402 /*
403 * 2nd step: Load the main header, validate it for correctness
404 */
405
406 m_pMainHeader->fixEndianness();
407
408 // Content type
409 switch (m_pMainHeader->iContentType) {
410 case DAFF_IMPULSE_RESPONSE:
411 case DAFF_MAGNITUDE_SPECTRUM:
412 case DAFF_PHASE_SPECTRUM:
413 case DAFF_MAGNITUDE_PHASE_SPECTRUM:
414 case DAFF_DFT_SPECTRUM:
415 break;
416
417 default:
418 // Invalid content type
419 tidyup();
420 return DAFF_FILE_CONTENT_TYPE_UNKOWN;
421 };
422
423 // Quantization
424 switch (m_pMainHeader->iQuantization) {
425 case DAFF_INT16:
426 case DAFF_INT24:
427 case DAFF_FLOAT32:
428 break;
429
430 default:
431 // Invalid quantization
432 tidyup();
433 return DAFF_FILE_QUANTIZATION_UNKOWN;
434 };
435
436 if (m_pMainHeader->iNumChannels < 1)
437 return DAFF_FILE_INVALID_MAIN_PARAMETER;
438
439 if (m_pMainHeader->iNumRecords < 1)
440 return DAFF_FILE_INVALID_MAIN_PARAMETER;
441
442 if (m_pMainHeader->iElementsPerRecord < 1)
443 return DAFF_FILE_INVALID_MAIN_PARAMETER;
444
445 if (m_pMainHeader->iAlphaPoints < 1)
446 return DAFF_FILE_ALPHA_ANGLES_INVALID;
447
448 // alpha start value must not be 360&deg;
449 if ((m_pMainHeader->fAlphaStart < 0.0f) || (m_pMainHeader->fAlphaStart >= 360.0f))
450 return DAFF_FILE_ALPHA_ANGLES_INVALID;
451
452 // alpha stop value may be 360&deg; indicating that the full alpha range is covered.
453 if ((m_pMainHeader->fAlphaEnd < 0.0f) || (m_pMainHeader->fAlphaEnd > 360.0f))
454 return DAFF_FILE_ALPHA_ANGLES_INVALID;
455
456 // Beta angle validation
457 if (m_pMainHeader->iBetaPoints < 1)
458 return DAFF_FILE_BETA_ANGLES_INVALID;
459
460 if (m_pMainHeader->fBetaStart > m_pMainHeader->fBetaEnd)
461 return DAFF_FILE_BETA_ANGLES_INVALID;
462
463 if ((m_pMainHeader->fBetaStart < 0.0f) || (m_pMainHeader->fBetaStart > 180.0f))
464 return DAFF_FILE_BETA_ANGLES_INVALID;
465
466 if ((m_pMainHeader->fBetaEnd < 0.0f) || (m_pMainHeader->fBetaEnd > 180.0f))
467 return DAFF_FILE_BETA_ANGLES_INVALID;
468
469 // Orientation
470 m_orientationDefault.fYawAngleDeg = m_pMainHeader->fOrientYaw;
471 m_orientationDefault.fPitchAngleDeg = m_pMainHeader->fOrientPitch;
472 m_orientationDefault.fRollAngleDeg = m_pMainHeader->fOrientRoll;
473 m_tTrans.setOrientation(m_orientationDefault);
474
475 return DAFF_NO_ERROR;
476}

478int DAFFReaderImpl::loadContentHeader()
479{
480 /*
481 * 3rd step: Load the content header, validate it for correctness
482 */
483
484 float* pfFreqs = 0;
485
486 switch (m_pMainHeader->iContentType) {
487 case DAFF_IMPULSE_RESPONSE:
488 m_pContentHeaderIR = static_cast<DAFFContentHeaderIR*>(m_pContentHeader);
489 m_pContentHeaderIR->fixEndianness();
490
491 if (m_pContentHeaderIR->fSamplerate < 0.0f)
492 return DAFF_FILE_CONTENT_INVALID_PARAMETER;
493
494 if ((m_pContentHeaderIR->iMaxEffectiveFilterLength < 0) ||
495 (m_pContentHeaderIR->iMaxEffectiveFilterLength > m_pMainHeader->iElementsPerRecord))
496 return DAFF_FILE_CONTENT_INVALID_PARAMETER;
497
498 if ((m_pContentHeaderIR->iMinFilterOffset < 0) ||
499 (m_pContentHeaderIR->iMinFilterOffset > m_pMainHeader->iElementsPerRecord))
500 return DAFF_FILE_CONTENT_INVALID_PARAMETER;
501
502 break;
503
504 case DAFF_MAGNITUDE_SPECTRUM:
505 m_pContentHeaderMS = static_cast<DAFFContentHeaderMS*>(m_pContentHeader);
506 m_pContentHeaderMS->fixEndianness();
507
508 if (m_pContentHeaderMS->iNumFreqs <= 0)
509 return DAFF_FILE_CONTENT_INVALID_PARAMETER;
510
511 pfFreqs = (float*)((char*)m_pContentHeader + 8);
512
513 // Fix the endianness of the frequency list
514 DAFF::le2se_4byte(pfFreqs, m_pContentHeaderMS->iNumFreqs);
515
516 // Load the (dynamically sized) frequency list
517 m_vfFreqs.resize(m_pContentHeaderMS->iNumFreqs);
518 for (int i = 0; i < m_pContentHeaderMS->iNumFreqs; i++)
519 m_vfFreqs[i] = pfFreqs[i];
520
521 break;
522
523 case DAFF_PHASE_SPECTRUM:
524 m_pContentHeaderPS = static_cast<DAFFContentHeaderPS*>(m_pContentHeader);
525 m_pContentHeaderPS->fixEndianness();
526
527 if (m_pContentHeaderPS->iNumFreqs <= 0)
528 return DAFF_FILE_CONTENT_INVALID_PARAMETER;
529
530 pfFreqs = (float*)((char*)m_pContentHeader + 8);
531
532 // Fix the endianness of the frequency list
533 DAFF::le2se_4byte(pfFreqs, m_pContentHeaderPS->iNumFreqs);
534
535 // Load the (dynamically sized) frequency list
536 m_vfFreqs.resize(m_pContentHeaderPS->iNumFreqs);
537 for (int i = 0; i < m_pContentHeaderPS->iNumFreqs; i++)
538 m_vfFreqs[i] = pfFreqs[i];
539
540 break;
541
542 case DAFF_MAGNITUDE_PHASE_SPECTRUM:
543 m_pContentHeaderMPS = static_cast<DAFFContentHeaderMPS*>(m_pContentHeader);
544 m_pContentHeaderMPS->fixEndianness();
545
546 if (m_pContentHeaderMPS->iNumFreqs <= 0)
547 return DAFF_FILE_CONTENT_INVALID_PARAMETER;
548
549 pfFreqs = (float*)((char*)m_pContentHeader + 8);
550
551 // Fix the endianness of the frequency list
552 DAFF::le2se_4byte(pfFreqs, m_pContentHeaderMPS->iNumFreqs);
553
554 // Load the (dynamically sized) frequency list
555 m_vfFreqs.resize(m_pContentHeaderMPS->iNumFreqs);
556 for (int i = 0; i < m_pContentHeaderMPS->iNumFreqs; i++)
557 m_vfFreqs[i] = pfFreqs[i];
558
559 break;
560
561 case DAFF_DFT_SPECTRUM:
562 m_pContentHeaderDFT = static_cast<DAFFContentHeaderDFT*>(m_pContentHeader);
563 m_pContentHeaderDFT->fixEndianness();
564
565 if ((m_pContentHeaderDFT->iNumDFTCoeffs != m_pContentHeaderDFT->iTransformSize) &&
566 (m_pContentHeaderDFT->iNumDFTCoeffs != floor(m_pContentHeaderDFT->iTransformSize / 2.0) + 1))
567 return DAFF_FILE_CONTENT_INVALID_PARAMETER;
568
569 break;
570 }
571
572 return DAFF_NO_ERROR;
573}

575int DAFFReaderImpl::loadRecordDescriptor()
576{
577 /*
578 * 4rd step: Load the record descriptor block
579 */
580
581 // Set access pointer and fix the endianness
582 switch (m_pMainHeader->iContentType) {
583 case DAFF_IMPULSE_RESPONSE:
584 // A single IR record channel desc is 4+4+4+8 Byte = 20 Bytes
585 m_iRecordChannelDescSize = sizeof(DAFFRecordChannelDescIR);
586 assert(m_iRecordChannelDescSize == 20);
587
588 // Fix endianness for channel descriptors and metadata index
589 for (int i = 0; i < m_pMainHeader->iNumRecords; i++) {
590 for (int c = 0; c < m_pMainHeader->iNumChannels; c++) {
591 DAFFRecordChannelDescIR* pDesc =
592 reinterpret_cast<DAFFRecordChannelDescIR*>(getRecordChannelDescPtr(i, c));
593 pDesc->fixEndianness();
594 }
595
596 DAFF::le2se_4byte(getRecordMetadataIndexPtr(i), 1);
597 };
598
599 break;
600
601 case DAFF_MAGNITUDE_SPECTRUM:
602 case DAFF_PHASE_SPECTRUM:
603 case DAFF_MAGNITUDE_PHASE_SPECTRUM:
604 case DAFF_DFT_SPECTRUM:
605 // All other content use a default record channel desc (MS/PS/MPS/DFT) which is 8 Bytes
606 m_iRecordChannelDescSize = sizeof(DAFFRecordChannelDescDefault);
607
608 // Fix endianness for channel descriptors and metadata index
609 for (int i = 0; i < m_pMainHeader->iNumRecords; i++) {
610 for (int c = 0; c < m_pMainHeader->iNumChannels; c++) {
611 DAFFRecordChannelDescDefault* pDesc =
612 reinterpret_cast<DAFFRecordChannelDescDefault*>(getRecordChannelDescPtr(i, c));
613 pDesc->fixEndianness();
614 }
615
616 DAFF::le2se_4byte(getRecordMetadataIndexPtr(i), 1);
617 };
618
619 break;
620 };
621
622 return DAFF_NO_ERROR;
623}

666 // Read all the DAFFMetadataImpl instances

668 size_t iBytesRead = 0;
669 char* pCurrentBuf = pMetadataBuf;
670 char* pEndOfMetaDataBuf = pCurrentBuf + (size_t)pMetadataFileBlock->ui64Size;
671 while (pCurrentBuf < pEndOfMetaDataBuf) {
672 pMetadataImpl = new DAFFMetadataImpl;
673 int iError = pMetadataImpl->load(pCurrentBuf, iBytesRead);
674 if ((iError != 0) || (iBytesRead == 0)) {
675 tidyup();
676 return iError;
677 }
678
679 m_vpMetadata.push_back(pMetadataImpl);
680
681 // Set Buffer to next DAFFMetaDataImpl instance
682 pCurrentBuf += iBytesRead;
683 }
684 }
685
686 return DAFF_NO_ERROR;
687}

689void DAFFReaderImpl::fixAngleRanges()
690{
691 // Important: If there is only one point in a dimension => Then there is no resolution
692 // TODO: Recheck if this is correct for non-full and full angular ranges
693
694 float fAlphaSpan;
695 if (m_pMainHeader->fAlphaEnd > m_pMainHeader->fAlphaStart)
696 fAlphaSpan = m_pMainHeader->fAlphaEnd - m_pMainHeader->fAlphaStart;
697 else
698 fAlphaSpan = 360 - m_pMainHeader->fAlphaStart + m_pMainHeader->fAlphaEnd;
699 float fBetaSpan = m_pMainHeader->fBetaEnd - m_pMainHeader->fBetaStart;
700
701 /*
702 * [fwe] Bugfix 2011-05-30
703 * Only if the full azimuth range is span, the formula
704 *
705 * m_fAlphaResolution = fAlphaSpan / m_pMainHeader->iAlphaPoints
706 *
707 * is valid. In case there is no full alpha coverage, the last point
708 * marks the end of the interval and does not count. Therefore:
709 *
710 * m_fAlphaResolution = fAlphaSpan / (m_pMainHeader->iAlphaPoints-1)
711 *
712 */
713
714 if (m_pMainHeader->iAlphaPoints > 1) {
715 if (fAlphaSpan == 360)
716 m_fAlphaResolution = fAlphaSpan / m_pMainHeader->iAlphaPoints;
717 else
718 m_fAlphaResolution = fAlphaSpan / (m_pMainHeader->iAlphaPoints - 1);
719 } else
720 m_fAlphaResolution = 0;
721
722 if (m_pMainHeader->iBetaPoints > 1)
723 m_fBetaResolution = fBetaSpan / (m_pMainHeader->iBetaPoints - 1);
724 else
725 m_fBetaResolution = 0;
726}

1087int DAFFReaderImpl::getRecordCoords(int iRecordIndex, int iView, float& fAngle1, float& fAngle2) const
1088{

1090 assert((iView == DAFF_DATA_VIEW) || (iView == DAFF_OBJECT_VIEW));
1091
1092 // Invalid record index - projection to get a valid index
1093 if ((iRecordIndex < 0) || (iRecordIndex >= m_pMainHeader->iNumRecords)) {
1094 // TODO: Discuss if this makes sense
1095 return -1;
1096 }
1097
1098 int iAlpha, iBeta;
1099 float fAlpha, fBeta;
1100
1101 bool bSouthPolePresent = false;
1102 if (m_pMainHeader->fBetaStart == 0.0f) // South pole present (single record here)
1103 bSouthPolePresent = true;
1104
1105 if (bSouthPolePresent) {
1106 if (iRecordIndex == 0) { // Hit south pole
1107 fAlpha = 0.0f;
1108 fBeta = 0.0f;
1109 } else { // somewhere inside the grid (with single value at south pole)
1110 iAlpha = (iRecordIndex - 1) % m_pMainHeader->iAlphaPoints;
1111 iBeta = 1 + (int)((iRecordIndex - 1) / m_pMainHeader->iAlphaPoints);
1112 fAlpha = m_pMainHeader->fAlphaStart + ((float)iAlpha * m_fAlphaResolution);
1113 fBeta = (float)iBeta * m_fBetaResolution;
1114 }
1115 } else { // somewhere inside the grid (without south pole present)
1116 iAlpha = iRecordIndex % m_pMainHeader->iAlphaPoints;
1117 iBeta = (int)(iRecordIndex / m_pMainHeader->iAlphaPoints);
1118 fAlpha = m_pMainHeader->fAlphaStart + ((float)iAlpha * m_fAlphaResolution);
1119 fBeta = m_pMainHeader->fBetaStart + ((float)iBeta * m_fBetaResolution);
1120 }
1121
1122 if (iView == DAFF_DATA_VIEW) {
1123 fAngle1 = fAlpha;
1124 fAngle2 = fBeta;
1125 } else {
1126 transformAnglesD2O(fAlpha, fBeta, fAngle1, fAngle2);
1127 }
1128
1129 return 0;
1130}

1132void DAFFReaderImpl::getNearestNeighbour(int iView, float fAngle1, float fAngle2, int& iRecordIndex,
1133 bool& bOutOfBounds) const
1134{
1135 assert((iView == DAFF_DATA_VIEW) || (iView == DAFF_OBJECT_VIEW));
1136
1137 float fAlpha;
1138 float fBeta;
1139
1140 if (iView == DAFF_DATA_VIEW) {
1141 fAlpha = fAngle1;
1142 fBeta = fAngle2;
1143 } else {
1144 // Transform the coordinates into the DSC
1145 transformAnglesO2D(fAngle1, fAngle2, fAlpha, fBeta);
1146 }
1147
1148 // Normalize the direction
1149 DAFFUtils::NormalizeDirection(DAFF_DATA_VIEW, fAlpha, fBeta, fAlpha, fBeta);
1150
1151 iRecordIndex = -1;
1152 bOutOfBounds = false;
1153
1154 int iAlphaIndex, iBetaIndex;
1155
1156 if (m_pMainHeader->iAlphaPoints == 1) {
1157 // Trivial case: Just one point
1158 iAlphaIndex = 0;
1159 bOutOfBounds = true;
1160 if (fAlpha == m_pMainHeader->fAlphaStart && fAlpha == std::fmod(m_pMainHeader->fAlphaEnd, 360.0f))
1161 bOutOfBounds = false; // Direct hit
1162 } else {
1163 if ((fAlpha >= m_pMainHeader->fAlphaStart) && (fAlpha <= m_pMainHeader->fAlphaEnd)) {
1164 // Within the covered alpha range
1165 iAlphaIndex = (int)roundf((fAlpha - m_pMainHeader->fAlphaStart) / m_fAlphaResolution);
1166 } else {
1167 // Outside the covered alpha range
1168 // Decide: Which is closer? Start boundary or end boundary?
1169 if (DAFF::anglef_mindiff_abs_0_360_DEG(m_pMainHeader->fAlphaStart, fAlpha) <=
1170 DAFF::anglef_mindiff_abs_0_360_DEG(m_pMainHeader->fAlphaEnd, fAlpha))
1171 iAlphaIndex = 0; // Start index
1172 else
1173 iAlphaIndex = (int)m_pMainHeader->iAlphaPoints - 1; // End index
1174
1175 bOutOfBounds = true;
1176 }
1177 }
1178
1179 if (m_pMainHeader->iBetaPoints == 1) {
1180 // Trivial case: Just one point
1181 iBetaIndex = 0;
1182 bOutOfBounds = true;
1183 if (fBeta == m_pMainHeader->fBetaEnd && fBeta == m_pMainHeader->fBetaStart)
1184 bOutOfBounds = false; // Direct hit
1185 } else {
1186 if ((fBeta >= m_pMainHeader->fBetaStart) && (fBeta <= m_pMainHeader->fBetaEnd)) {
1187 // Within the covered beta range
1188 iBetaIndex = (int)roundf((fBeta - m_pMainHeader->fBetaStart) / m_fBetaResolution);
1189 } else {
1190 // Outside the covered beta range
1191 // Decide: Which is closer? Start boundary or end boundary?
1192 if (std::abs(m_pMainHeader->fBetaStart - fBeta) <= std::abs(m_pMainHeader->fBetaEnd - fBeta))
1193 iBetaIndex = 0; // Start index
1194 else
1195 iBetaIndex = (int)m_pMainHeader->iBetaPoints - 1; // End index
1196
1197 bOutOfBounds = true;
1198 }
1199 }
1200
1201 // Calculate index
1202 if (m_pMainHeader->fBetaStart == 0.0f) { // South pole present: increment by one (single record at poles)
1203 if (iBetaIndex == 0) { // Hit south pole
1204 iRecordIndex = 0;
1205 } else {
1206 if ((iBetaIndex == m_pMainHeader->iBetaPoints - 1) &&
1207 (m_pMainHeader->fBetaEnd == 180.0f)) // Hit north pole
1208 iRecordIndex = 1 + (iBetaIndex - 1) * m_pMainHeader->iAlphaPoints;
1209 else
1210 iRecordIndex = 1 + (iBetaIndex - 1) * m_pMainHeader->iAlphaPoints + iAlphaIndex;
1211 }
1212 } else {
1213 if ((iBetaIndex == m_pMainHeader->iBetaPoints - 1) && (m_pMainHeader->fBetaEnd == 180.0f)) // Hit north pole
1214 iRecordIndex = iBetaIndex * m_pMainHeader->iAlphaPoints;
1215 else
1216 iRecordIndex = iBetaIndex * m_pMainHeader->iAlphaPoints + iAlphaIndex;
1217 }
1218
1219 return;
1220}

1222void DAFFReaderImpl::getCell(int iView, const float fAngle1, const float fAngle2, DAFFQuad& qIndices) const
1223{
1224 // switch to data view
1225 float fAlpha = fAngle1;
1226 float fBeta = fAngle2;
1227 if (iView == DAFF_OBJECT_VIEW)
1228 transformAnglesO2D(fAngle1, fAngle2, fAlpha, fBeta);
1229 DAFFUtils::NormalizeDirection(DAFF_DATA_VIEW, fAlpha, fBeta, fAlpha, fBeta);
1230
1231 // south pole with full sphere covered is a problem (but algorithm holds for north pole)
1232 if (fBeta == 0.0f && coversFullSphere()) { // return the south pole 4 times
1233 getNearestNeighbour(DAFF_DATA_VIEW, 0.0f, 0.0f, qIndices.iIndex1);
1234 qIndices.iIndex2 = qIndices.iIndex1;
1235 qIndices.iIndex3 = qIndices.iIndex1;
1236 qIndices.iIndex4 = qIndices.iIndex1;
1237 return;
1238 }
1239
1240 // grid angles
1241 float fAlpha1 = 0, fAlpha2 = 0, fAlpha3 = 0, fAlpha4 = 0;
1242 float fBeta1 = 0, fBeta2 = 0, fBeta3 = 0, fBeta4 = 0;
1243
1244 fAlpha1 = fAlpha - fmodf(fAlpha, getAlphaResolution());
1245 fAlpha2 = fAlpha1;
1246 fAlpha3 = fAlpha + getAlphaResolution() - fmodf(fAlpha, getAlphaResolution());
1247 fAlpha4 = fAlpha3;
1248
1249 fBeta1 = fBeta - fmodf(fBeta, getBetaResolution());
1250 fBeta2 = fBeta + getBetaResolution() - fmodf(fBeta, getBetaResolution());
1251 fBeta3 = fBeta2;
1252 fBeta4 = fBeta1;
1253
1254 // Upper beta angles are not allowed to overrun the north pole
1255 if (fBeta2 > 180.0f) {
1256 fBeta2 = 180.0f;
1257 fBeta3 = 180.0f;
1258 }
1259
1260 // Normalize & get indices
1261 DAFFUtils::NormalizeDirection(DAFF_DATA_VIEW, fAlpha1, fBeta1, fAlpha1, fBeta1);
1262 DAFFUtils::NormalizeDirection(DAFF_DATA_VIEW, fAlpha2, fBeta2, fAlpha2, fBeta2);
1263 DAFFUtils::NormalizeDirection(DAFF_DATA_VIEW, fAlpha3, fBeta3, fAlpha3, fBeta3);
1264 DAFFUtils::NormalizeDirection(DAFF_DATA_VIEW, fAlpha4, fBeta4, fAlpha4, fBeta4);
1265
1266 getNearestNeighbour(DAFF_DATA_VIEW, fAlpha1, fBeta1, qIndices.iIndex1);
1267 getNearestNeighbour(DAFF_DATA_VIEW, fAlpha2, fBeta2, qIndices.iIndex2);
1268 getNearestNeighbour(DAFF_DATA_VIEW, fAlpha3, fBeta3, qIndices.iIndex3);
1269 getNearestNeighbour(DAFF_DATA_VIEW, fAlpha4, fBeta4, qIndices.iIndex4);
1270}
