25/* Transformation functions
26
27 These transformations perform angular conversions between two spherical coordinate systems,
28 which are rotated against each other by yaw-pitch-roll convention. So basically what we do here,
29 is to express a yaw-pitch-roll rotation within spherical coordinates - this is not trivial.
30
31 The formulas for the rotation can be derived by introducing a cartesian coordinate system
32 (e.g. right-handed OpenGL, where Z is up) as an intermediate format. Then rotations are applied
33 in the cartesian domainn. Finally we transform this result back into spherical coordinates.
34
35 Many of the terms for the conversion only depend on the rotation angles itself (yaw, pitch and roll).
36 Therefore we calculate these constants once and cache them for later reuse.
37
38 For the angular calculations double precision floating points are necessary to maintain
39 a low numerical error.
40
41
42 Operation count:
43
44 4 trigenometric (2 sin, 2 cos)
45 2 muls (subexpressions)
46 9 muls (calculation)
47 6 adds (calculation)
48 2 inverse trigenometric (1 atan2, 1 asin)
49 -----------------------------------------
50 Total: 6 trigenometric + 17 madds
51
52 [Frank Wefers, 2010-02-01]
53
54 */

63void DAFFSCTransform::transformOSC2DSC(double azimuth_in, double elevation_in, double& alpha_out,
64 double& beta_out) const
65{
66 // Internally in this method we use radians (not degrees)
67 double ai = DAFFUtils::grad2rad(azimuth_in), ei = DAFFUtils::grad2rad(elevation_in);
68 double ao, eo;
69
70 // Base terms
71 double sa = sin(ai), ca = cos(ai);
72 double se = sin(ei), ce = cos(ei);
73
74 // Common subexpressions
75 double sa_ce = sa * ce;
76 double ca_ce = ca * ce;
77
78 ao = atan2(m_const.t1 * sa_ce - m_const.t2 * se + m_const.t3 * ca_ce,
79 m_const.t4 * sa_ce - m_const.t5 * se + m_const.t6 * ca_ce);
80 eo = asin(m_const.t7 * sa_ce + m_const.t8 * se + m_const.t9 * ca_ce);
81
82 alpha_out = DAFFUtils::rad2grad(ao);
83 beta_out = DAFFUtils::rad2grad(eo) + 90.0F; // Translate into DSC (b=e+90)
84}

94void DAFFSCTransform::transformDSC2OSC(double alpha_in, double beta_in, double& azimuth_out,
95 double& elevation_out) const
96{
97 // Internally in this method we use radians (not degrees)
98 double ai = DAFFUtils::grad2rad(alpha_in),
99 ei = DAFFUtils::grad2rad(beta_in - 90.0F); // Translate into OSC (e=b-90)
100 double ao, eo;
101
102 // Base terms
103 double sa = sin(ai), ca = cos(ai);
104 double se = sin(ei), ce = cos(ei);
105
106 // Common subexpressions
107 double sa_ce = sa * ce;
108 double ca_ce = ca * ce;
109
110 ao = atan2(m_const.t1 * sa_ce + m_const.t7 * se + m_const.t4 * ca_ce,
111 m_const.t3 * sa_ce + m_const.t9 * se + m_const.t6 * ca_ce);
112 eo = -asin(m_const.t2 * sa_ce - m_const.t8 * se + m_const.t5 * ca_ce);
113
114 azimuth_out = DAFFUtils::rad2grad(ao);
115 elevation_out = DAFFUtils::rad2grad(eo);
116}

118void DAFFSCTransform::RotationConstants::Init(double yaw, double pitch, double roll)
119{
120 double y = DAFFUtils::grad2rad(yaw);
121 double p = DAFFUtils::grad2rad(pitch);
122 double r = DAFFUtils::grad2rad(roll);
123
124 double sy = sin(y), cy = cos(y);
125 double sp = sin(p), cp = cos(p);
126 double sr = sin(r), cr = cos(r);
127
128 t1 = (cy * cr - sy * sp * sr);
129 t2 = (cy * sr + sy * sp * cr);
130 t3 = sy * cp;
131 t4 = (-sy * cr - cy * sp * sr);
132 t5 = (-sy * sr + cy * sp * cr);
133 t6 = cy * cp;
134 t7 = cp * sr;
135 t8 = cp * cr;
136 t9 = sp;
137}
