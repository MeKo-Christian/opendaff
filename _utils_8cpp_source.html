39float anglef_mindiff_0_2PI(float alpha, float beta)
40{
41 float gamma = anglef_proj_0_2PI(beta) - anglef_proj_0_2PI(alpha);
42 if (gamma >= 0)
43 return (gamma <= PI_F ? gamma : gamma - 2 * PI_F);
44 else
45 return (gamma >= -PI_F ? gamma : gamma + 2 * PI_F);
46}

118// Swap with temporary variable
119#define SWAP(i, j) \
120 tmp = p[i]; \
121 p[i] = p[j]; \
122 p[j] = tmp;

124void byteswap_2byte(void* src, size_t count)
125{
126 // In: | B1 | B0 |
