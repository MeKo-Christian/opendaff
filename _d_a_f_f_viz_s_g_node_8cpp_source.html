129bool SGNode::AddChildNode(DAFFViz::SGNode* pChild)
130{
131 /*
132 * Wichtig: Der Knoten darf nur dann hinzugefügt werden, falls er nicht
133 * bereits Unterknoten in einem anderen (Teil)baum ist. Dies
134 * stellt sicher, das jeder Knoten höchstens in einem Teilbaum
135 * enthalten sein ist. Der Test erfolgt über die Wurzel.
136 */
137
138 assert(pChild != NULL);
139 if (pChild == NULL)
140 return false;
141
142 // Ausschließen das man einen Knoten zu sich selbst verknüpft
143 assert(pChild != this);
144 if (pChild == this)
145 return false;
146
147 // Child already has a parent (root) node
148 assert(pChild->IsRoot());
149 if (!pChild->IsRoot())
150 return false;
151
152 // Intern den Unterbaum verknüpfen
153 m_vpChildNodes.push_back(pChild);
154
155 // Beim Kinderknoten den Vaterknoten vermerken
156 pChild->m_pParentNode = this;
157
158 // Assemblies verknüpfen
159 DAFFVIZ_LOCK_VTK;
160 m_pNodeAssembly->AddPart(pChild->m_pNodeAssembly);
161 DAFFVIZ_UNLOCK_VTK;
162
163 return true;
164}

235void SGNode::SetOrientationYPR(double yaw, double pitch, double roll)
236{
237 double* pos = m_pNodeAssembly->GetPosition();
238
239 double r = DAFFUtils::grad2rad(roll);
240 double p = DAFFUtils::grad2rad(pitch);
241 double y = DAFFUtils::grad2rad(yaw);
242
243 // Translation matrix: to origin
244 vtkSmartPointer<vtkMatrix4x4> mOrigin = vtkSmartPointer<vtkMatrix4x4>::New();
245 mOrigin->Identity();
246 mOrigin->SetElement(0, 3, -pos[0]);
247 mOrigin->SetElement(1, 3, -pos[1]);
248 mOrigin->SetElement(2, 3, -pos[2]);
249
250 // Translation matrix: to object position
251 vtkSmartPointer<vtkMatrix4x4> mPosition = vtkSmartPointer<vtkMatrix4x4>::New();
252 mPosition->Identity();
253 mPosition->SetElement(0, 3, pos[0]);
254 mPosition->SetElement(1, 3, pos[1]);
255 mPosition->SetElement(2, 3, pos[2]);
256
257 /*
258 * Roll: homogenous rotation matrix for -Z axis
259 *
260 *
261 * cos(alpha) sin(alpha) 0 0
262 * -sin(alpha) cos(alpha) 0 0
263 * 0 0 1 0
264 * 0 0 0 1
265 */
266 vtkSmartPointer<vtkMatrix4x4> mRoll = vtkSmartPointer<vtkMatrix4x4>::New();
267 mRoll->Identity();
268 mRoll->SetElement(0, 0, cos(r));
269 mRoll->SetElement(0, 1, sin(r));
270 mRoll->SetElement(1, 0, -sin(r));
271 mRoll->SetElement(1, 1, cos(r));
272
273 /*
274 * Pitch: homogenous rotation matrix for +X axis
275 *
276 * 1 0 0 0
277 * 0 cos(alpha) -sin(alpha) 0
278 * 0 sin(alpha) cos(alpha) 0
279 * 0 0 0 1
280 */
281 vtkSmartPointer<vtkMatrix4x4> mPitch = vtkSmartPointer<vtkMatrix4x4>::New();
282 mPitch->Identity();
283 mPitch->SetElement(1, 1, cos(p));
284 mPitch->SetElement(1, 2, -sin(p));
285 mPitch->SetElement(2, 1, sin(p));
286 mPitch->SetElement(2, 2, cos(p));
287
288 /*
289 * Yaw: homogenous rotation matrix for +Y axis
290 *
291 * cos(alpha) 0 sin(alpha) 0
292 * 0 1 0 0
293 * -sin(alpha) 0 cos(alpha) 0
294 * 0 0 0 1
295 */
296 vtkSmartPointer<vtkMatrix4x4> mYaw = vtkSmartPointer<vtkMatrix4x4>::New();
297 mYaw->Identity();
298 mYaw->SetElement(0, 0, cos(y));
299 mYaw->SetElement(0, 2, sin(y));
300 mYaw->SetElement(2, 0, -sin(y));
301 mYaw->SetElement(2, 2, cos(y));
302
303
304 // Compose transforms
305 // (Order: Origin, Roll, Pitch, Yaw, Position)
306 vtkSmartPointer<vtkTransform> transform = vtkSmartPointer<vtkTransform>::New();
307 transform->Concatenate(mPosition);
308 transform->Concatenate(mYaw);
309 transform->Concatenate(mPitch);
310 transform->Concatenate(mRoll);
311 transform->Concatenate(mOrigin);
312
313 // Apply new transformation
314 DAFFVIZ_LOCK_VTK;
315 m_pNodeAssembly->SetUserTransform(transform);
316 DAFFVIZ_UNLOCK_VTK;
317
318 return;
319}

321void SGNode::SetOrientationVU(double vx, double vy, double vz, double ux, double uy, double uz)
322{
323 double* pos = m_pNodeAssembly->GetPosition();
324
325 // Translation matrix: to origin
326 vtkSmartPointer<vtkMatrix4x4> mOrigin = vtkSmartPointer<vtkMatrix4x4>::New();
327 mOrigin->Identity();
328 mOrigin->SetElement(0, 3, -pos[0]);
329 mOrigin->SetElement(1, 3, -pos[1]);
330 mOrigin->SetElement(2, 3, -pos[2]);
331
332 // Translation matrix: to object position
333 vtkSmartPointer<vtkMatrix4x4> mPosition = vtkSmartPointer<vtkMatrix4x4>::New();
334 mPosition->Identity();
335 mPosition->SetElement(0, 3, pos[0]);
336 mPosition->SetElement(1, 3, pos[1]);
337 mPosition->SetElement(2, 3, pos[2]);
338
339 // Cross product
340 double zx = uz * vy - uy * vz;
341 double zy = ux * vz - uz * vx;
342 double zz = uy * vx - ux * vy;
343
344 vtkSmartPointer<vtkMatrix4x4> mViewUp = vtkSmartPointer<vtkMatrix4x4>::New();
345 mViewUp->Identity();
346 mViewUp->SetElement(0, 0, zx);
347 mViewUp->SetElement(1, 0, zy);
348 mViewUp->SetElement(2, 0, zz);
349 mViewUp->SetElement(0, 1, ux);
350 mViewUp->SetElement(1, 1, uy);
351 mViewUp->SetElement(2, 1, uz);
352 mViewUp->SetElement(0, 2, -vx);
353 mViewUp->SetElement(1, 2, -vy);
354 mViewUp->SetElement(2, 2, -vz);
355
356 // Compose transforms
357 // (Order: Origin, Roll, Pitch, Yaw, Position)
358 vtkSmartPointer<vtkTransform> transform = vtkSmartPointer<vtkTransform>::New();
359 transform->Concatenate(mPosition);
360 transform->Concatenate(mViewUp);
361 transform->Concatenate(mOrigin);
362
363 // Apply new transformation
364 DAFFVIZ_LOCK_VTK;
365 m_pNodeAssembly->SetUserTransform(transform);
366 DAFFVIZ_UNLOCK_VTK;
367
368 return;
369}

371void SGNode::GetOrientationVU(double& vx, double& vy, double& vz, double& ux, double& uy, double& uz)
372{
373 vtkSmartPointer<vtkMatrix4x4> pM = m_pNodeAssembly->GetUserMatrix();
374
375 std::vector<double> vdView(4);
376 std::vector<double> vdDefaultView(4);
377 vdDefaultView[2] = -1.0f;
378 vdDefaultView[3] = 1.0f;
379 if (pM) {
380 vdView[0] = *pM->MultiplyDoublePoint(&vdDefaultView[0]);
381 vx = vdView[0];
382 vy = vdView[1];
383 vz = vdView[2];
384 } else {
385 for (size_t i = 0; i < 4; i++)
386 vdView[i] = vdDefaultView[i];
387 }
388
389 std::vector<double> vdUp(4);
390 std::vector<double> vdDefaultUp(4);
391 vdDefaultUp[1] = vdDefaultUp[3] = 1.0f;
392 if (pM) {
393 vdUp[0] = *pM->MultiplyDoublePoint(&vdDefaultUp[0]);
394 ux = vdUp[0];
395 uy = vdUp[1];
396 uz = vdUp[2];
397 } else {
398 for (size_t i = 0; i < 4; i++)
399 vdUp[i] = vdDefaultUp[i];

403void SGNode::GetOrientationYPR(double& dYawDeg, double& dPitchDeg, double& dRollDeg)
404{
405 double dEps = DAFF::EPSILON_D;
406 double dPi = DAFF::PI_D;
407
408 double vx, vy, vz, ux, uy, uz, yaw, pitch, roll;
409 GetOrientationVU(vx, vy, vz, ux, uy, uz);
410
411 if (vy >= (1 - dEps)) {
412 yaw = atan2(ux, uz);
413 pitch = dPi;
414 roll = 0;
415
416 dYawDeg = DAFFUtils::rad2grad(yaw);
417 dPitchDeg = DAFFUtils::rad2grad(pitch);
418 dRollDeg = DAFFUtils::rad2grad(roll);
419
420 return;
421 }
422
423 if (vy <= -(1 - dEps)) {
424 yaw = atan2(-ux, -uz);
425 pitch = -dPi;
426 roll = 0;
427
428 dYawDeg = DAFFUtils::rad2grad(yaw);
429 dPitchDeg = DAFFUtils::rad2grad(pitch);
430 dRollDeg = DAFFUtils::rad2grad(roll);
431
432 return;
433 }
434
435 yaw = atan2(-vx, -vz);
436 pitch = asin(vy);
437
438 double zy = vz * ux - vx * uz;
439
440 if ((uy <= dEps) && (uy >= -dEps)) {
441 // y-component of cross production v x u
442 double zy = vz * ux - vx * uz;
443 roll = (zy <= 0 ? dPi : -dPi);
444
445 dYawDeg = DAFFUtils::rad2grad(yaw);
446 dPitchDeg = DAFFUtils::rad2grad(pitch);
447 dRollDeg = DAFFUtils::rad2grad(roll);
448
449 return;
450 }
451
452 // Hint: cos(pitch) = cos( arcsin(vy) ) = sqrt(1-vy^2)
453 double cp = sqrt(1 - vy * vy);
454 double uy_by_cp = uy / cp;
455 if (std::fabs(uy_by_cp - 1) < dEps)
456 roll = 0;
457 else
458 roll = (zy <= 0 ? acos(uy_by_cp) : -acos(uy_by_cp));
459
460 dYawDeg = DAFFUtils::rad2grad(yaw);
461 dPitchDeg = DAFFUtils::rad2grad(pitch);
462 dRollDeg = DAFFUtils::rad2grad(roll);
463
464 return;
465}
