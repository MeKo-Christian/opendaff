166void DAFFTransformerIR2DFT::transform()
167{
168 // Discard previous DFT representation
169 clear();
170
171 int iRecords(0);
172 int iChannels(0);
173 int iFilterLength(0);
174
175 m_fOverallMagnitudePeak = 0;
176
177 // Create new output content instance
178 if (m_pInputContent) {
179 m_pOutputContent = new DAFFContentDFTRealization(this, m_pInputContent);
180
181 m_iNumDFTCoeffs = m_pOutputContent->getNumDFTCoeffs();
182
183 // Number of bytes per DFT spectrum. Aligned on 16-bytes.
184 m_iElementSize = m_iNumDFTCoeffs * 2;
185 int iElementSizeBytes = m_iElementSize * sizeof(float);
186 int n = iElementSizeBytes / 16;
187 if ((iElementSizeBytes % 16) > 0)
188 n++;
189 iElementSizeBytes = n * 16;
190
191 // Allocate buffer for DFT spectra
192 iRecords = m_pInputContent->getProperties()->getNumberOfRecords();
193 iChannels = m_pInputContent->getProperties()->getNumberOfChannels();
194
195 m_pfBuf = static_cast<float*>(DAFF::malloc_aligned16(iRecords * iChannels * iElementSizeBytes));
196 }
197
198 // Transform the contents
199 if (m_pOutputContent) {
200 iFilterLength = m_pInputContent->getFilterLength();
201 float* pfInputData = static_cast<float*>(DAFF::malloc_aligned16(iFilterLength * sizeof(float)));
202 float* pfWindow = static_cast<float*>(DAFF::malloc_aligned16(iFilterLength * sizeof(float)));
203
204 fftwf_plan oFFTPlan =
205 fftwf_plan_dft_r2c_1d(iFilterLength, pfInputData, reinterpret_cast<fftwf_complex*>(m_pfBuf), FFTW_ESTIMATE);
206
207 // Create window function
208 switch (m_iWindowFunction) {
209 case DAFF_WINDOW_HANN: {
210 // TODO:
211 } break;
212 }
213
214 // Written using one loop for effective parallelization
215 for (int n = 0; n < iRecords * iChannels; n++) {
216 int iRecord = n / iChannels;
217 int iChannel = n % iChannels;
218
219 // Fetch input data
220 m_pInputContent->getFilterCoeffs(iRecord, iChannel, pfInputData);
221
222 // Apply window function
223 if (m_iWindowFunction != DAFF_WINDOW_NONE)
224 for (int i = 0; i < iFilterLength; i++)
225 pfInputData[i] *= pfWindow[i];
226
227 // Transform into frequency-domain
228 float* pfOutputData = m_pfBuf + (iRecord * iChannels + iChannel) * m_iElementSize;
229 fftwf_execute_dft_r2c(oFFTPlan, pfInputData, reinterpret_cast<fftwf_complex*>(pfOutputData));
230
231 // Determine maximum
232 for (int i = 0; i < m_iNumDFTCoeffs; i++) {
233 float fReal = pfOutputData[2 * i + 0];
234 float fImag = pfOutputData[2 * i + 1];
235 float fMagSq = fReal * fReal + fImag * fImag;
236 m_fOverallMagnitudePeak = (std::max)(m_fOverallMagnitudePeak, fMagSq);
237 }
238 }
239
240 DAFF::free_aligned16(pfInputData);
241 DAFF::free_aligned16(pfWindow);
242
243 fftwf_destroy_plan(oFFTPlan);
244
245 m_fOverallMagnitudePeak = sqrt(m_fOverallMagnitudePeak);
246 }
247}

254int DAFFTransformerIR2DFT::getDFTCoeff(int iRecordIndex, int iChannel, int iDFTCoeff, float& fReal, float& fImag) const
255{
256 if (!m_pInputContent)
257 return -1;
258
259 int iRecords = m_pInputContent->getProperties()->getNumberOfRecords();
260 int iChannels = m_pInputContent->getProperties()->getNumberOfChannels();
261
262 assert((iRecordIndex >= 0) && (iRecordIndex < iRecords));
263 assert((iChannel >= 0) && (iChannel < iChannels));
264 assert((iDFTCoeff >= 0) && (iDFTCoeff < m_iNumDFTCoeffs));
265
266 if ((iRecordIndex < 0) || (iRecordIndex >= iRecords))
267 return -1;
268 if ((iChannel < 0) || (iChannel >= iChannels))
269 return -1;
270 if ((iDFTCoeff < 0) || (iDFTCoeff > m_iNumDFTCoeffs))
271 return -1;
272
273 float* pfData = m_pfBuf + (iRecordIndex * iChannels + iChannel) * m_iElementSize;
274
275 fReal = pfData[2 * iDFTCoeff + 0];
276 fImag = pfData[2 * iDFTCoeff + 1];
277
278 return 0;
279}
