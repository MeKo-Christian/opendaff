134 public:

136 float fPitchAngleDeg;
137 float fRollAngleDeg;
138
140 inline DAFFOrientationYPR() : fYawAngleDeg(0), fPitchAngleDeg(0), fRollAngleDeg(0) {};
141
143
148 inline DAFFOrientationYPR(float fInitYawAngleDeg, float fInitPitchAngleDeg, float fInitRollAngleDeg)
149 : fYawAngleDeg(fInitYawAngleDeg), fPitchAngleDeg(fInitPitchAngleDeg), fRollAngleDeg(fInitRollAngleDeg) {};
150
152
155 inline std::string toString() const
156 {
157 std::stringstream ss;
158 // Note: \xF8 => Degree sign
159 ss << "( Y" << fYawAngleDeg << "\xF8, P" << fPitchAngleDeg << "\xF8, R" << fRollAngleDeg << "\xF8 )";
160 return ss.str();
161 };
