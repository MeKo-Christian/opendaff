153void CarpetPlot::InitCarpetMesh()
154{
155 const DAFFProperties* pProps = m_pContentIR->getProperties();
156
157 // --= carpet grid points and faces =--
158
159 int iNumPointsX, iIndexMaxX;
160 if (m_iFixedAngle == BETA_FIXED) {
161 iNumPointsX = pProps->getAlphaPoints();
162 if (pProps->coversFullAlphaRange())
163 iIndexMaxX = iNumPointsX + 1; // when we cover full sphere, the last row equals the first one
164 else
165 iIndexMaxX = iNumPointsX;
166 } else {
167 iNumPointsX = pProps->getBetaPoints();
168 iIndexMaxX = iNumPointsX;
169 }
170
171 int iNumPointsY = m_pContentIR->getFilterLength();
172
173 vtkIdType quadface[4];
174 vtkSmartPointer<vtkPoints> points = vtkSmartPointer<vtkPoints>::New();
175 vtkSmartPointer<vtkCellArray> cells = vtkSmartPointer<vtkCellArray>::New();
176 for (int i = 0; i < iIndexMaxX; i++) {
177 for (int j = 0; j < iNumPointsY; j++) {
178 if ((i != iIndexMaxX - 1) && (j != iNumPointsY - 1)) { // every time except from first and last row/column
179 quadface[0] = (i + 1) * iNumPointsY + j + 1;
180 quadface[1] = (i + 1) * iNumPointsY + j;
181 quadface[2] = (i)*iNumPointsY + j;
182 quadface[3] = (i)*iNumPointsY + j + 1;
183 for (int k = 0; k < 4; k++)
184 assert((quadface[k] >= 0) && (quadface[k] < iIndexMaxX * iNumPointsY));
185 cells->InsertNextCell(4, quadface);
186 }
187 vtkIdType iPointiD = i * iNumPointsY + j;
188 double dX = 2.0f * double(i) / double(iNumPointsX) - 1.0f;
189 double dY = 2.0f * double(j) / double(iNumPointsY) - 1.0f;
190 double dZ = 0.0f;
191 points->InsertPoint(iPointiD, dX, dY, dZ);
192 }
193 }
194
195 m_pCarpetPolyData->SetPoints(points);
196 m_pCarpetPolyData->SetPolys(cells);
197
198 SetScalars();
199}

374void CarpetPlot::SetScalars()
375{
376 // TODO: asserts
377 assert(m_pContentIR != NULL);
378 assert(m_dMin < m_dMax);
379
380 vtkSmartPointer<vtkFloatArray> pIRWarpArray = vtkSmartPointer<vtkFloatArray>::New();
381 pIRWarpArray->SetName("DAFFIRWarpData");
382 std::vector<float> vCoeffs(m_pContentIR->getFilterLength());
383 if (m_iFixedAngle == BETA_FIXED) {
384 int iNumPoints = m_pContentIR->getProperties()->getAlphaPoints();
385 float fStart = m_pContentIR->getProperties()->getAlphaStart();
386 float fRange = m_pContentIR->getProperties()->getAlphaEnd() - fStart;
387 for (int i = 0; i <= iNumPoints; i++) {
388 // Pull filter coefficients
389 int index = 0;
390 m_pContentIR->getNearestNeighbour(DAFF_DATA_VIEW, (float)i / (float)iNumPoints * fRange + fStart, m_fAngle,
391 index);
392 m_pContentIR->getFilterCoeffs(index, m_iChannel, &vCoeffs[0]);
393 float coeff = 0.0f;
394 for (int j = 0; j < m_pContentIR->getFilterLength(); j++) {
395 coeff = vCoeffs[j];
396
397 // Check weather decibel scaling is activated
398 if (m_iScaling == SCALING_DECIBEL) {
399 // CAVE: we take absolut value instead of square
400 coeff = (float)factor2decibel((double)fabs(coeff));
401
402 // Decibel boundaries
403 float DECIBEL_LOWER = -100;
404 if (m_dMin >= 0)
405 DECIBEL_LOWER = factor2decibel(m_dMin);
406 float DECIBEL_UPPER = factor2decibel(m_dMax);
407
408 // Normalize the range into the interval [0,1]
409 // Limit the boundaries
410 coeff = std::max(coeff, DECIBEL_LOWER);
411 coeff = std::min(coeff, DECIBEL_UPPER);
412 // clamp to [0 1]
413 coeff =
414 1.0 / (DECIBEL_UPPER - DECIBEL_LOWER) * coeff + DECIBEL_LOWER / (DECIBEL_LOWER - DECIBEL_UPPER);
415 } else {
416 // Limit the boundaries
417 coeff = std::max(coeff, m_dMin);
418 coeff = std::min(coeff, m_dMax);
419 // clamp to [0 1]
420 coeff = (coeff - m_dMin) / (m_dMax - m_dMin);
421 }
422 // store value in array
423 pIRWarpArray->InsertNextTuple1(coeff);
424 }
425 }
426 } else {
427 int iNumPoints = m_pContentIR->getProperties()->getBetaPoints();
428 float fStart = m_pContentIR->getProperties()->getBetaStart();
429 float fRange = m_pContentIR->getProperties()->getBetaEnd() - fStart;
430 for (int i = 0; i < m_pContentIR->getProperties()->getBetaPoints(); i++) {
431 // Pull filter coefficients
432 int index = 0;
433 m_pContentIR->getNearestNeighbour(DAFF_DATA_VIEW, m_fAngle, (float)i / (float)iNumPoints * fRange + fStart,
434 index);
435 m_pContentIR->getFilterCoeffs(index, m_iChannel, &vCoeffs[0]);
436 float coeff = 0.0f;
437 for (int j = 0; j < m_pContentIR->getFilterLength(); j++) {
438 coeff = vCoeffs[j];
439
440 // Check weather decibel scaling is activated
441 if (m_iScaling == SCALING_DECIBEL) {
442 // CAVE: we take absolut value instead of square
443 coeff = (float)factor2decibel((double)fabs(coeff));
444
445 // Decibel boundaries
446 float DECIBEL_LOWER = -100;
447 if (m_dMin >= 0)
448 DECIBEL_LOWER = factor2decibel(m_dMin);
449 float DECIBEL_UPPER = factor2decibel(m_dMax);
450
451 // Normalize the range into the interval [0,1]
452 // Limit the boundaries
453 coeff = std::max(coeff, DECIBEL_LOWER);
454 coeff = std::min(coeff, DECIBEL_UPPER);
455 // clamp to [0 1]
456 coeff =
457 1.0 / (DECIBEL_UPPER - DECIBEL_LOWER) * coeff + DECIBEL_LOWER / (DECIBEL_LOWER - DECIBEL_UPPER);
458 } else {
459 // Limit the boundaries
460 coeff = std::max(coeff, m_dMin);
461 coeff = std::min(coeff, m_dMax);
462 // clamp to [0 1]
463 coeff = (coeff - m_dMin) / (m_dMax - m_dMin);
464 }
465 // store value in array
466 pIRWarpArray->InsertNextTuple1(coeff);
467 }
468 }
469 }
470
471 // Assign scalars to points
472 m_pCarpetPolyData->GetPointData()->SetScalars(pIRWarpArray);
473
474 // move plot down so it fits to the axes
475 updatePlotOffset();
476 UpdateProbe();
477
478 return;
479}

481double CarpetPlot::factor2decibel(double x) const
482{
483 assert(x >= 0);
484 // Lower boundary is -100 dB
485 if (x <= 0.0000000001)
486 return -100;
487 return 10 * log(x) / log((double)10);
488}
