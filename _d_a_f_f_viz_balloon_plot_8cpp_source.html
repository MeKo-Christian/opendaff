60void BalloonPlot::init()
61{
62 const DAFFProperties* pProps = m_pContent->getProperties();
63
64 switch (pProps->getContentType()) {
65 case DAFF_DFT_SPECTRUM:
66 m_iNumFrequencies = dynamic_cast<const DAFFContentDFT*>(m_pContent)->getNumDFTCoeffs();
67 break;
68
69 case DAFF_MAGNITUDE_SPECTRUM:
70 case DAFF_MAGNITUDE_PHASE_SPECTRUM:
71 case DAFF_PHASE_SPECTRUM:
72 m_iNumFrequencies = dynamic_cast<const DAFFContentMS*>(m_pContent)->getNumFrequencies();
73 break;
74 }
75
76 bool bSouthPole = pProps->getBetaStart() == 0;
77 bool bNorthPole = pProps->getBetaEnd() == 180;
78 bool bAzimuthWrap = (pProps->getAlphaStart() == 0) && (pProps->getAlphaEnd() == 360);
79
80
81 // --= Sphere grid =--
82
83 vtkSmartPointer<vtkPoints> points = vtkSmartPointer<vtkPoints>::New();
84 m_pNormals = vtkSmartPointer<vtkDoubleArray>::New();
85 m_pNormals->SetNumberOfComponents(3);
86 for (int i = 0; i < pProps->getNumberOfRecords(); i++) {
87 // TODO: Topologies: Rings, Patches, etc.
88 float phi, theta;
89 m_pContent->getRecordCoords(i, DAFF_OBJECT_VIEW, phi, theta);
90
91 // Compute Cartesian points
92 double x, y, z;
93 sph2cart(phi, theta, x, y, z);
94 points->InsertPoint(i, x, y, z);
95 double n[3] = {-x, -y, -z}; // define flipped normals for warping
96 m_pNormals->InsertNextTuple(n);
97 }
98
99
100 // --= Faces =--
101
102 vtkSmartPointer<vtkCellArray> cells = vtkSmartPointer<vtkCellArray>::New();
103 vtkIdType quadface[4];
104 vtkIdType triangleface[3];
105
106 int iNumPoints = pProps->getNumberOfRecords();
107 int iAziPoints = pProps->getAlphaPoints();
108 int iElePoints = pProps->getBetaPoints();
109
110 int iBodyEleRings = iElePoints - 1;
111 int iAziCount = (bAzimuthWrap ? iAziPoints : iAziPoints - 1);
112 if (bSouthPole)
113 iBodyEleRings--;
114 if (bNorthPole)
115 iBodyEleRings--;
116 int iBodyOffset = 0;
117 if (bSouthPole)
118 iBodyOffset = 1;
119
120 // if there is only one azimuth point there would be no surface to draw, so lets draw a line
121 if ((iAziPoints == 1) || (iElePoints == 1)) {
122 if ((iAziPoints == 1) && (iElePoints == 1)) {
123 // we will not plot a single point
124 } else {
125 // add a point at center
126 points->InsertPoint(iNumPoints, 0, 0, 0);
127
128 for (int i = 0; i < iNumPoints; i++) {
129 triangleface[0] = i;
130 triangleface[1] = i + 1;
131 triangleface[2] = iNumPoints;
132
133 for (int k = 0; k < 3; k++)
134 assert((triangleface[k] >= 0) && (triangleface[k] < iNumPoints + 1));
135
136 cells->InsertNextCell(3, triangleface);
137 }
138 }
139 } else {
140 // draw surfaces
141 if (bSouthPole) {
142 /*
143 * If there is a south pole, this has index 0.
144 * The next record, one elevation step above has then index 1.
145 */
146
147 iBodyOffset = 1;
148 for (int j = 0; j < iAziCount; j++) {
149 triangleface[0] = 0;
150 triangleface[1] = iBodyOffset + j;
151 triangleface[2] = iBodyOffset + (j + 1) % iAziPoints;
152
153 for (int k = 0; k < 3; k++)
154 assert((triangleface[k] >= 0) && (triangleface[k] < iNumPoints));
155
156 cells->InsertNextCell(3, triangleface);
157 }
158 };
159
160 // Body
161 for (int i = 0; i < iBodyEleRings; i++) {
162 for (int j = 0; j < iAziCount; j++) {
163 quadface[0] = iBodyOffset + i * iAziPoints + j;
164 quadface[1] = iBodyOffset + i * iAziPoints + (j + 1) % iAziPoints;
165 quadface[2] = iBodyOffset + (i + 1) * iAziPoints + (j + 1) % iAziPoints;
166 quadface[3] = iBodyOffset + (i + 1) * iAziPoints + j;
167
168 for (int k = 0; k < 4; k++)
169 assert((quadface[k] >= 0) && (quadface[k] < iNumPoints));
170
171 cells->InsertNextCell(4, quadface);
172 }
173 }
174
175 // North pole
176
177 if (bNorthPole) {
178 /*
179 * If there is a north pole, this has the last index.
180 * The next record, one elevation step above has then index 1.
181 */
182
183 for (int j = 0; j < iAziCount; j++) {
184 triangleface[0] = iNumPoints - 1;
185 triangleface[1] = iBodyOffset + iBodyEleRings * iAziPoints + j;
186 triangleface[2] = iBodyOffset + iBodyEleRings * iAziPoints + (j + 1) % iAziPoints;
187
188 for (int k = 0; k < 3; k++)
189 assert((triangleface[k] >= 0) && (triangleface[k] < pProps->getNumberOfRecords()));
190
191 cells->InsertNextCell(3, triangleface);
192 }
193 }
194 }
195
196
197 // --= VTK stuff =--
198 m_pPlotPolydata = vtkSmartPointer<vtkPolyData>::New();
199 m_pPlotPolydata->SetPoints(points);
200 m_pPlotPolydata->SetPolys(cells);
201
202 // Set scalars of polydata
203 SetScalars();
204 m_pPlotPolydata->GetPointData()->SetActiveScalars("magnitudes");
205
206 // Warp sphere
207 m_pWarp = vtkSmartPointer<vtkWarpScalar>::New();
208 m_pWarp->SetInputData(m_pPlotPolydata);
209 m_pWarp->UseNormalOff();
210
211 m_pMapper = vtkSmartPointer<vtkPolyDataMapper>::New();
212 m_pMapper->SetScalarModeToUsePointFieldData();
213
214 if (m_bWarp)
215 EnableWarp();
216 else
217 DisableWarp();
218
219 SetUsePhaseAsColor(m_bUsePhaseAsColor);
220
221 m_pPlotActor = vtkSmartPointer<vtkActor>::New();
222 m_pPlotActor->SetMapper(m_pMapper);
223
224 m_pPlotActor->GetProperty()->SetInterpolationToGouraud();
225
226 AddActor(m_pPlotActor);
227
228 // Probe
229
230 vtkSmartPointer<vtkPoints> points2 = vtkSmartPointer<vtkPoints>::New();
231 points2->SetNumberOfPoints(2);
232 points2->InsertPoint(0, 0, 0, 0);
233 points2->InsertPoint(1, 0, -1, 0);
234
235 vtkSmartPointer<vtkCellArray> cells2 = vtkSmartPointer<vtkCellArray>::New();
236 vtkSmartPointer<vtkLine> line = vtkSmartPointer<vtkLine>::New();
237 line->GetPointIds()->SetId(0, 0);
238 line->GetPointIds()->SetId(1, 1);
239 cells2->InsertNextCell(line);
240
241 vtkSmartPointer<vtkPolyData> polydata = vtkSmartPointer<vtkPolyData>::New();
242 polydata->SetPoints(points2);
243 polydata->SetLines(cells2);
244
245 // Create a mapper and actor
246 vtkSmartPointer<vtkPolyDataMapper> mapper = vtkSmartPointer<vtkPolyDataMapper>::New();
247 mapper->SetInputData(polydata);
248
249 m_pProbe = vtkSmartPointer<vtkActor>::New();
250 m_pProbe->SetMapper(mapper);
251 m_pProbe->VisibilityOff();
252
253 AddActor(m_pProbe);
254
255 // label the probe
256 m_pProbeLabel = vtkSmartPointer<vtkVectorText>::New();
257 std::ostringstream s;
258 s << "test";
259 m_pProbeLabel->SetText(s.str().c_str());
260
261 vtkSmartPointer<vtkPolyDataMapper> mapper2 = vtkSmartPointer<vtkPolyDataMapper>::New();
262 mapper2->AddInputConnection(m_pProbeLabel->GetOutputPort());
263
264 m_pLabel = vtkSmartPointer<vtkFollower>::New();
265 m_pLabel->SetMapper(mapper2);
266 m_pLabel->SetScale(0.03);
267 m_pLabel->SetPosition(0.02, -1.01, 0);
268 m_pLabel->SetOrientation(-90, 0, -90);
269 m_pLabel->GetProperty()->SetOpacity(0.8);
270 m_pLabel->VisibilityOff();
271
272 AddActor(m_pLabel);
273}

275void BalloonPlot::UpdateProbe()
276{
277 if (!m_pProbe)
278 return;
279
280 double* pos = m_pProbe->GetPosition();
281 // double r = grad2rad();
282 double p = DAFFUtils::grad2rad(m_dProbeBeta);
283 double y = DAFFUtils::grad2rad(m_dProbeAlpha);
284
285 // Translation matrix: to origin
286 vtkSmartPointer<vtkMatrix4x4> mOrigin = vtkSmartPointer<vtkMatrix4x4>::New();
287 mOrigin->Identity();
288 mOrigin->SetElement(0, 3, -pos[0]);
289 mOrigin->SetElement(1, 3, -pos[1]);
290 mOrigin->SetElement(2, 3, -pos[2]);
291
292 // Translation matrix: to object position
293 vtkSmartPointer<vtkMatrix4x4> mPosition = vtkSmartPointer<vtkMatrix4x4>::New();
294 mPosition->Identity();
295 mPosition->SetElement(0, 3, pos[0]);
296 mPosition->SetElement(1, 3, pos[1]);
297 mPosition->SetElement(2, 3, pos[2]);
298
299 /*
300 * Roll: homogenous rotation matrix for -Z axis
301 *
302 *
303 * cos(alpha) sin(alpha) 0 0
304 * -sin(alpha) cos(alpha) 0 0
305 * 0 0 1 0
306 * 0 0 0 1
307 */
308 // vtkSmartPointer< vtkMatrix4x4 >* mRoll = vtkSmartPointer< vtkMatrix4x4 >::New();
309 // mRoll->Identity();
310 // mRoll->SetElement(0,0, cos(r));
311 // mRoll->SetElement(0,1, sin(r));
312 // mRoll->SetElement(1,0, -sin(r));
313 // mRoll->SetElement(1,1, cos(r));
314
315 /*
316 * Pitch: homogenous rotation matrix for +X axis
317 *
318 * 1 0 0 0
319 * 0 cos(alpha) -sin(alpha) 0
320 * 0 sin(alpha) cos(alpha) 0
321 * 0 0 0 1
322 */
323 vtkSmartPointer<vtkMatrix4x4> mPitch = vtkSmartPointer<vtkMatrix4x4>::New();
324 mPitch->Identity();
325 mPitch->SetElement(1, 1, cos(p));
326 mPitch->SetElement(1, 2, -sin(p));
327 mPitch->SetElement(2, 1, sin(p));
328 mPitch->SetElement(2, 2, cos(p));
329
330 /*
331 * Yaw: homogenous rotation matrix for +Y axis
332 *
333 * cos(alpha) 0 sin(alpha) 0
334 * 0 1 0 0
335 * -sin(alpha) 0 cos(alpha) 0
336 * 0 0 0 1
337 */
338 vtkSmartPointer<vtkMatrix4x4> mYaw = vtkSmartPointer<vtkMatrix4x4>::New();
339 mYaw->Identity();
340 mYaw->SetElement(0, 0, cos(y));
341 mYaw->SetElement(0, 2, sin(y));
342 mYaw->SetElement(2, 0, -sin(y));
343 mYaw->SetElement(2, 2, cos(y));
344
345
346 // Compose transforms
347 // (Order: Origin, Roll, Pitch, Yaw, Position)
348 vtkSmartPointer<vtkTransform> transform = vtkSmartPointer<vtkTransform>::New();
349 transform->Concatenate(mPosition);
350 transform->Concatenate(mYaw);
351 transform->Concatenate(mPitch);
352 // transform->Concatenate(mRoll);
353 transform->Concatenate(mOrigin);
354
355 // Apply new transformation
356 m_pProbe->SetUserTransform(transform);
357 m_pLabel->SetUserTransform(transform);
358
359 // update label
360 assert(m_pContent != 0);
361 double val = 0;
362 int i;
363 m_pContent->getNearestNeighbour(DAFF_DATA_VIEW, m_dProbeAlpha, m_dProbeBeta, i);
364 std::ostringstream s;
365 const DAFFContentDFT* pContentDFT = NULL;
366 const DAFFContentMS* pContentMS = NULL;
367 const DAFFContentMPS* pContentMPS = NULL;
368
369 float fMag, fPhase, fReal, fImag;
370
371 switch (m_pContent->getProperties()->getContentType()) {
372 case DAFF_DFT_SPECTRUM:
373 pContentDFT = dynamic_cast<const DAFFContentDFT*>(m_pContent);
374 pContentDFT->getDFTCoeff(i, m_iChannel, m_iFrequency, fReal, fImag);
375 fMag = sqrt(fReal * fReal + fImag * fImag);
376 if (fReal != 0.0) {
377 fPhase = atan(fImag / fReal);
378 if (fReal < 0.0)
379 if (fImag < 0.0)
380 fPhase -= DAFF::PI_F;
381 else
382 fPhase += DAFF::PI_F;
383 } else {
384 if (fImag < 0.0)
385 fPhase = -DAFF::HALF_PI_F;
386 else
387 fPhase = DAFF::HALF_PI_F;
388 }
389 if (m_iScaling == SCALING_DECIBEL) {
390 fMag = FactorToDecibel(fMag);
391 s << "Real: " << fReal << ", Imag: " << fImag << "\n Mag: " << fMag << "db, Phase:" << fPhase;
392 } else
393 s << "Real: " << fReal << ", Imag: " << fImag << "\n Mag: " << fMag << ", Phase:" << fPhase;
394 break;
395
396 case DAFF_MAGNITUDE_SPECTRUM:
397 pContentMS = dynamic_cast<const DAFFContentMS*>(m_pContent);
398 pContentMS->getMagnitude(i, m_iChannel, m_iFrequency, fMag);
399 if (m_iScaling == SCALING_DECIBEL) {
400 fMag = FactorToDecibel(fMag);
401 s << fMag << "db";
402 } else
403 s << fMag;
404 break;
405
406 case DAFF_MAGNITUDE_PHASE_SPECTRUM:
407 pContentMPS = dynamic_cast<const DAFFContentMPS*>(m_pContent);
408 pContentMPS->getMagnitude(i, m_iChannel, m_iFrequency, fMag);
409 pContentMPS->getPhase(i, m_iChannel, m_iFrequency, fPhase);
410 if (m_iScaling == SCALING_DECIBEL) {
411 fMag = FactorToDecibel(fMag);
412 s << "Mag: " << fMag << "db, Phase: " << fPhase;
413 } else
414 s << "Mag: " << fMag << ", Phase: " << fPhase;
415 break;
416 }
417 m_pProbeLabel->SetText(s.str().c_str());
418
419 return;
420}

524void BalloonPlot::SetScalars()
525{
526 assert(m_iFrequency >= 0 && m_iFrequency < m_iNumFrequencies);
527 assert(m_pContent != NULL);
528 assert(m_dMin < m_dMax);
529
530 const DAFFContentDFT* pContentDFT = NULL;
531 const DAFFContentMS* pContentMS = NULL;
532 const DAFFContentMPS* pContentMPS = NULL;
533
534 switch (m_pContent->getProperties()->getContentType()) {
535 case DAFF_DFT_SPECTRUM:
536 pContentDFT = dynamic_cast<const DAFFContentDFT*>(m_pContent);
537 break;
538
539 case DAFF_MAGNITUDE_SPECTRUM:
540 pContentMS = dynamic_cast<const DAFFContentMS*>(m_pContent);
541 break;
542
543 case DAFF_MAGNITUDE_PHASE_SPECTRUM:
544 pContentMPS = dynamic_cast<const DAFFContentMPS*>(m_pContent);
545 break;
546 }
547
548 vtkSmartPointer<vtkFloatArray> pMagArray = vtkSmartPointer<vtkFloatArray>::New();
549
550 pMagArray->SetName("magnitudes");
551 vtkSmartPointer<vtkFloatArray> pPhArray = vtkSmartPointer<vtkFloatArray>::New();
552 pPhArray->SetName("phases");
553
554 float fMag = 0.0f;
555 float fPhase = 0.0f;
556 float fAbsoluteMax = 0.0f;
557 double dMax = 1.0f;
558
559 // get the normalization range
560 if (pContentDFT) {
561 if (m_iScaling == SCALING_DECIBEL) {
562 if (m_bNormalizeFreqsIndiv)
563 fAbsoluteMax = getMagnitudeMaximum();
564 else
565 fAbsoluteMax = pContentDFT->getOverallMagnitudeMaximum();
566 } else {
567 if (m_bNormalizeFreqsIndiv)
568 dMax = getMagnitudeMaximum();
569 else
570 dMax = pContentDFT->getOverallMagnitudeMaximum();
571 }
572
573 } else if (pContentMS) {
574 if (m_bNormalizeFreqsIndiv)
575 dMax = getMagnitudeMaximum();
576 else
577 dMax = pContentMS->getOverallMagnitudeMaximum();
578 } else if (pContentMPS) {
579 if (m_bNormalizeFreqsIndiv)
580 dMax = getMagnitudeMaximum();
581 else
582 dMax = pContentMPS->getOverallMagnitudeMaximum();
583 }
584
585 for (int i = 0; i < m_pContent->getProperties()->getNumberOfRecords(); i++) {
586 // Get the magnitude value
587
588 if (pContentDFT) // ----------------
589 {
590 float fReal, fImag;
591 pContentDFT->getDFTCoeff(i, m_iChannel, m_iFrequency, fReal, fImag);
592 fMag = sqrt(fReal * fReal + fImag * fImag);
593 if (fReal != 0.0) {
594 fPhase = atan(fImag / fReal);
595 if (fReal < 0.0)
596 if (fImag < 0.0)
597 fPhase -= DAFF::PI_F;
598 else
599 fPhase += DAFF::PI_F;
600 } else {
601 if (fImag < 0.0)
602 fPhase = -DAFF::HALF_PI_F;
603 else
604 fPhase = DAFF::HALF_PI_F;
605 }
606
607 // Check weather decibel scaling is activated
608 if (m_iScaling == SCALING_DECIBEL) {
609 assert(fAbsoluteMax != NULL);
610 fMag = FactorToDecibel(fMag / fAbsoluteMax);
611
612 float DECIBEL_LOWER;
613 float DECIBEL_UPPER;
614
615 // Decibel boundaries
616 // float DECIBEL_DELTA = 30; // FIXME: should not be hard coded -> GUI
617 if (m_bUseCustomRange) {
618 DECIBEL_LOWER = FactorToDecibel(m_dMin);
619 DECIBEL_UPPER = FactorToDecibel(m_dMax);
620 } else {
621 DECIBEL_LOWER = FactorToDecibel(0.0);
622 DECIBEL_UPPER = FactorToDecibel(fAbsoluteMax);
623 }
624
625 // Limit the lower boundary
626 fMag = std::max(fMag, DECIBEL_LOWER);
627 fMag = std::min(fMag, DECIBEL_UPPER);
628
629 // Normalize the range into the interval [0,1]
630 fMag = 1 / (DECIBEL_UPPER - DECIBEL_LOWER) * fMag + DECIBEL_LOWER / (DECIBEL_LOWER - DECIBEL_UPPER);
631 } else {
632 if (m_bNormalize || m_bNormalizeFreqsIndiv) {
633 assert(dMax != NULL && fMag <= dMax);
634 // Normalize the range into the interval [0, 1]
635 fMag = fMag / dMax;
636 } else if (m_bUseCustomRange) {
637 fMag = std::max(fMag, m_dMin);
638 fMag = std::min(fMag, m_dMax);
639 // Normalize the range into the interval [0,1]
640 fMag = (fMag - m_dMin) / (m_dMax - m_dMin);
641 }
642 }
643
644 } else if (pContentMS) // ----------------
645 {
646 pContentMS->getMagnitude(i, m_iChannel, m_iFrequency, fMag);
647
648 if (m_bNormalize || m_bNormalizeFreqsIndiv) {
649 assert(dMax != NULL && fMag <= dMax);
650 // Normalize the range into the interval [0, 1]
651 fMag = fMag / dMax;
652 } else if (m_bUseCustomRange) {
653 fMag = std::max(fMag, m_dMin);
654 fMag = std::min(fMag, m_dMax);
655 // Normalize the range into the interval [0,1]
656 fMag = (fMag - m_dMin) / (m_dMax - m_dMin);
657 }
658 }
659
660 else if (pContentMPS) {
661 pContentMPS->getMagnitude(i, m_iChannel, m_iFrequency, fMag);
662 pContentMPS->getPhase(i, m_iChannel, m_iFrequency, fPhase);
663
664 if (m_bNormalize) {
665 assert(dMax != NULL && fMag <= dMax);
666 // Normalize the range into the interval [0, 1]
667 fMag = fMag / dMax;
668 } else if (m_bUseCustomRange) {
669 fMag = std::max(fMag, m_dMin);
670 fMag = std::min(fMag, m_dMax);
671 // Normalize the range into the interval [0,1]
672 fMag = (fMag - m_dMin) / (m_dMax - m_dMin);
673 }
674 }
675
676 // Store (inverted) value into scalars array
677 pMagArray->InsertNextTuple1(1.0 - fMag);
678 pPhArray->InsertNextTuple1(fPhase);
679 }
680
681 // Assign scalars to points
682 // m_pPlotPolydata->GetPointData()->SetScalars(pMagArray);
683 m_pPlotPolydata->GetPointData()->AddArray(pMagArray);
684 m_pPlotPolydata->GetPointData()->AddArray(pPhArray);
685
686 UpdateProbe();
687 return;
688}

690void BalloonPlot::sph2cart(double phi, double theta, double& x, double& y, double& z)
691{
692 x = -sin((theta + 90) * DAFF::PI_F / 180.0) * sin(phi * DAFF::PI_F / 180.0);
693 y = -cos((theta + 90) * DAFF::PI_F / 180.0);
694 z = -sin((theta + 90) * DAFF::PI_F / 180.0) * cos(phi * DAFF::PI_F / 180.0);
695}

713float BalloonPlot::getMagnitudeMaximum() const
714{
715 assert(m_iFrequency >= 0 && m_iFrequency < m_iNumFrequencies);
716 assert(m_pContent != NULL);
717 assert(m_dMin < m_dMax);
718
719 const DAFFContentDFT* pContentDFT = NULL;
720 const DAFFContentMS* pContentMS = NULL;
721 const DAFFContentMPS* pContentMPS = NULL;
722
723 switch (m_pContent->getProperties()->getContentType()) {
724 case DAFF_DFT_SPECTRUM:
725 pContentDFT = dynamic_cast<const DAFFContentDFT*>(m_pContent);
726 break;
727
728 case DAFF_MAGNITUDE_SPECTRUM:
729 pContentMS = dynamic_cast<const DAFFContentMS*>(m_pContent);
730 break;
731
732 case DAFF_MAGNITUDE_PHASE_SPECTRUM:
733 pContentMPS = dynamic_cast<const DAFFContentMPS*>(m_pContent);
734 break;
735 }
736
737 float fMax = std::numeric_limits<float>::min();
738 float fMag;
739 if (pContentDFT) {
740 float fReal, fImag;
741 for (int i = 0; i < pContentDFT->getNumDFTCoeffs(); i++) {
742 pContentDFT->getDFTCoeff(i, m_iChannel, m_iFrequency, fReal, fImag);
743 fMag = std::sqrtf(std::powf(fReal, 2) + std::powf(fImag, 2));
744 if (fMag > fMax)
745 fMax = fMag;
746 }
747 } else if (pContentMS) {
748 for (int i = 0; i < pContentMS->getProperties()->getNumberOfRecords(); i++) {
749 pContentMS->getMagnitude(i, m_iChannel, m_iFrequency, fMag);
750 if (fMag > fMax)
751 fMax = fMag;
752 }
753 } else if (pContentMPS) {
754 for (int i = 0; i < pContentMPS->getProperties()->getNumberOfRecords(); i++) {
755 pContentMPS->getMagnitude(i, m_iChannel, m_iFrequency, fMag);
756 if (fMag > fMax)
757 fMax = fMag;
758 }
759 }
760
761 return fMax;
762}
