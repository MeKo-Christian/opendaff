24// Helper function: Convert float => std::string (fixed format, but without trailing zeros)
25std::string DAFFUtils::Float2StrNice(float f, int precision, bool showpos, int leadingzeros)
26{
27 // First convert to fixed format (with enough digits)
28 std::stringstream ss;
29 if (showpos)
30 ss << (f < 0.0f ? "-" : "+");
31 ss << std::fixed << std::setprecision(precision);
32 if (leadingzeros > 0)
33 ss << std::setw(leadingzeros + precision + 1) << std::setfill('0');
34 if (showpos)
35 ss << fabs(f);
36 else
37 ss << f;
38 std::string s = ss.str();
39
40 // Then remove trailing zeros...
41 size_t n = s.length();
42 while ((n > 0) && (s[n - 1] == '0'))
43 n--;
44
45 // Special case: We reached the . => Remove this as well...
46 if ((n > 1) && (s[n - 1] == '.'))
47 n--;
48
49 return s.substr(0, n);
50}

80std::string DAFFUtils::StrDirection(int iView, double dAngle1, double dAngle2, int precision, int leadingzeros)
81{
82 std::stringstream ss;
83
84 switch (iView) {
85 case DAFF_DATA_VIEW:
86 ss << std::fixed << "( A" << Double2StrNice(dAngle1, precision, false, leadingzeros) << "\xF8, B"
87 << Double2StrNice(dAngle2, precision, false, leadingzeros) << "\xF8 )";
88 return ss.str();

90
91 case DAFF_OBJECT_VIEW:

94 return ss.str();
95 break;
96
97 default:
98 // Invalid view
99 assert(false);
100 return "";
101 }
102 // we will never reach this point
103 return "";
104}

238void DAFFUtils::NormalizeDirection(int iView, float fAngle1In, float fAngle2In, float& fAngle1Out, float& fAngle2Out)
239{
240 const float EPSILON = 0.00001F; // 10^-5 &deg;
241
242 if (iView == DAFF_DATA_VIEW) {
243 /*
244 * - Alpha always within [0&deg;, +360&deg;)
245 * - Beta always within [0&deg;, +180&deg;]
246 * - At poles (beta=0&deg;|beta=180&deg;) default: alpha = 0
247 */
248
249 float fAlpha = fAngle1In;
250 float fBeta = fmodf(fAngle2In, 360.0f);
251
252 if (fBeta > 180.0f) {
253 fAlpha += 180.0f;
254 fBeta -= 180.0f;
255 }
256
257 fAlpha = fmodf(fAlpha, 360.0f);
258 if (fAlpha < 0.0f)
259 fAlpha += 360.0f;
260
261 // Note: We use thresholds here to work on wrongly rounded angles also
262 if ((std::abs(fBeta) <= EPSILON) || // Beta == 0&deg;
263 (std::abs(fBeta - 180.0f) <= EPSILON)) // Beta == 180&deg;
264 fAlpha = 0.0f;
265
266 fAngle1Out = roundf(fAlpha * 1000.0f);
267 fAngle1Out /= 1000.0f;
268 fAngle2Out = roundf(fBeta * 1000.0f);
269 fAngle2Out /= 1000.0f;
270
271 return;
272 }
273
274 if (iView == DAFF_OBJECT_VIEW) {
275 /*
276 * - Azimuth always within (-180&deg;, +180&deg;]
277 * - Elevation always within [-90&deg;, +90&deg;]
278 * - At poles (elevation=+/-90&deg;) default: azimuth = 0
279 */
280
281 float fAzimuth = fAngle1In;
282 float fElevation = fmodf(fAngle2In, 360.0F);
283
284 if ((fElevation > 90.0F) && (fElevation < 270.0F)) {
285 fAzimuth += 180.0F;
286 fElevation = 180.0F - fElevation;
287 }
288
289 // Project azimuth angle into (-180&deg;, +180&deg;]
290 fAzimuth = fmodf(fAzimuth + 180.0F, 360.0F) - 180.0F;
291 if (fAzimuth == -180.0F)
292 fAzimuth = +180.0F;
293
294 // Note: We use thresholds here to work on wrongly rounded angles also
295 if (std::abs(fElevation - 90.0f) <= EPSILON) // Elevation == 90&deg;
296 fElevation = 90.0f;
297 if (std::abs(fElevation + 90.0f) <= EPSILON) // Elevation == -90&deg;
298 fElevation = -90.0f;
299
300 fAngle1Out = roundf(fAzimuth * 1000.0f);
301 fAngle1Out /= 1000.0f;
302 fAngle2Out = roundf(fElevation * 1000.0f);
303 fAngle2Out /= 1000.0f;
304
305 return;
306 }
307
308 // Invalid view
309 assert(false);
310}
