50int DAFFMetadataImpl::load(void* pData, size_t& iBytesRead)
51{
52 char* p = (char*)pData;
53
54 // Read the number of keys
55 int* piNumKeys = (int*)p;
56 DAFF::le2se_4byte(piNumKeys, 1);
57 p += 4;
58
59 int* piDatatype;
60 int* piValue;
61 double* pdValue;
62 std::string sValue;
63
64 // Read the keys
65 for (int i = 0; i < *piNumKeys; i++) {
66 // Datatype
67 piDatatype = (int*)p;
68 DAFF::le2se_4byte(piDatatype, 1);
69 p += 4;
70
71 // Key name
72 std::string sKey(p);
73 p += sKey.length() + 1;
74
75 // Value
76 switch (*piDatatype) {
77 case DAFF_BOOL:
78 piValue = (int*)p;
79 DAFF::le2se_4byte(piValue, 1);
80 p += 4;
81
82 insertKey(sKey, new DAFFMetadataKeyBool(*piValue != 0));
83 break;
84
85 case DAFF_INT:
86 piValue = (int*)p;
87 DAFF::le2se_4byte(piValue, 1);
88 p += 4;
89
90 insertKey(sKey, new DAFFMetadataKeyInt(*piValue));
91 break;
92
93 case DAFF_FLOAT:
94 pdValue = (double*)p;
95 DAFF::le2se_8byte(pdValue, 1);
96 p += 8;
97
98 insertKey(sKey, new DAFFMetadataKeyFloat(*pdValue));
99 break;
100
101 case DAFF_STRING:
102 sValue = p;
103 p += sValue.length() + 1;
104
105 insertKey(sKey, new DAFFMetadataKeyString(sValue));
106 break;
107
108 default:
109 return DAFF_FILE_UNKOWN_METADATA_TYPE;
110 }
111 }
112
113 iBytesRead = (size_t)(p - (char*)pData);
114 return 0;
115}

147std::string DAFFMetadataImpl::getKeyString(const std::string& sKey) const
148{
149 const DAFFMetadataKey* pKey = findKey(sKey);
150 assert(pKey != NULL);
151
152 if (pKey == NULL)
153 return "";
154
155 std::stringstream ss;
156 switch (pKey->m_iType) {
157 case DAFF_BOOL:
158 return (dynamic_cast<const DAFFMetadataKeyBool*>(pKey)->m_bValue ? "yes" : "no");
159
160 case DAFF_INT:
161 ss << dynamic_cast<const DAFFMetadataKeyInt*>(pKey)->m_iValue;
162 return ss.str();
163
164 case DAFF_FLOAT:
165 ss << dynamic_cast<const DAFFMetadataKeyFloat*>(pKey)->m_dValue;
166 return ss.str();
167
168 case DAFF_STRING:
169 return dynamic_cast<const DAFFMetadataKeyString*>(pKey)->m_sValue;
170
171 default:
172 // This will never be reached, but satisfies the compiler warning...
173 return "";
174 }

235const DAFFMetadataKey* DAFFMetadataImpl::findKey(const std::string& sKey) const
236{
237 // Empty strings are no valid key names
238 if (sKey.empty())
239 return NULL;
240
241 // Convert the keyname to upper case
242 std::string sKeyUpper(sKey);

244
245 // Search in the keymap

247 return (cit == m_mKeys.end() ? NULL : cit->second);
